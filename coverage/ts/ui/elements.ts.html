/**
 * Get a specific element by ID with type casting
 * @param id - Element ID
 * @returns Element or null
 */
export function getElement<T extends HTMLElement>(id: string): T | null {
    return document.getElementById(id) as T | null;
}

/**
 * Show or hide an element
 * @param element - Element to show/hide
 * @param show - Whether to show or hide
 * @param displayType - CSS display value when shown (default: 'flex')
 */
export function showElement(
    element: HTMLElement | null,
    show: boolean,
    displayType: string = 'flex'
): void {
    if (element) {
        element.style.display = show ? displayType : 'none';
    }
}

/**
 * Set element text content safely
 * @param element - Element to update
 * @param text - Text to set
 */
export function setElementText(element: HTMLElement | null, text: string): void {
    if (element) {
        element.textContent = text;
    }
}

/**
 * Set element class name
 * @param element - Element to update
 * @param className - Class name to set
 */
export function setElementClass(element: HTMLElement | null, className: string): void {
    if (element) {
        element.className = className;
    }
}

/**
 * Toggle element class
 * @param element - Element to update
 * @param className - Class name to toggle
 * @param force - Optional force add/remove
 */
export function toggleElementClass(
    element: HTMLElement | null,
    className: string,
    force?: boolean
): void {
    if (element) {
        element.classList.toggle(className, force);
    }
}

/**
 * Set button disabled state
 * @param button - Button element
 * @param disabled - Disabled state
 */
export function setButtonDisabled(button: HTMLButtonElement | null, disabled: boolean): void {
    if (button) {
        button.disabled = disabled;
    }
}

/**
 * Set input value
 * @param input - Input element
 * @param value - Value to set
 */
export function setInputValue(input: HTMLInputElement | HTMLTextAreaElement | null, value: string): void {
    if (input) {
        input.value = value;
    }
}

export default {
    initializeElements,
    getElement,
    showElement,
    setElementText,
    setElementClass,
    toggleElementClass,
    setButtonDisabled,
    setInputValue
};
